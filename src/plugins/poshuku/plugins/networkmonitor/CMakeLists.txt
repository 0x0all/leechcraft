INCLUDE_DIRECTORIES (${POSHUKU_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR})
SET (NETWORKMONITOR_SRCS
	networkmonitor.cpp
	)
SET (NETWORKMONITOR_HEADERS
	networkmonitor.h
	)
SET (NETWORKMONITOR_FORMS
	networkmonitor.ui
	)
SET (NETWORKMONITOR_RESOURCES networkmonitorresources.qrc)
SET (NETWORKMONITOR_TRANSLATIONS
	leechcraft_poshuku_networkmonitor_ru_RU.ts
	leechcraft_poshuku_networkmonitor_uk_UA.ts
	)
SET (NETWORKMONITOR_COMPILED_TRANSLATIONS
	${CMAKE_CURRENT_SOURCE_DIR}/leechcraft_poshuku_networkmonitor_ru_RU.qm
	${CMAKE_CURRENT_SOURCE_DIR}/leechcraft_poshuku_networkmonitor_uk_UA.qm
	)
QT4_WRAP_CPP (NETWORKMONITOR_MOC_SRCS ${NETWORKMONITOR_HEADERS})
QT4_WRAP_UI (NETWORKMONITOR_UIS_H ${NETWORKMONITOR_FORMS})
QT4_ADD_RESOURCES (NETWORKMONITOR_RCCS ${NETWORKMONITOR_RESOURCES})
ADD_CUSTOM_COMMAND (OUTPUT
	${NETWORKMONITOR_COMPILED_TRANSLATIONS}
	COMMAND "${QT_LRELEASE_EXECUTABLE}" ${NETWORKMONITOR_TRANSLATIONS}
	DEPENDS ${NETWORKMONITOR_TRANSLATIONS}
	WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
	)
ADD_LIBRARY (leechcraft_poshuku_networkmonitor SHARED
	${NETWORKMONITOR_COMPILED_TRANSLATIONS}
	${NETWORKMONITOR_SRCS}
	${NETWORKMONITOR_MOC_SRCS}
	${NETWORKMONITOR_UIS_H}
	${NETWORKMONITOR_RCCS}
	)
TARGET_LINK_LIBRARIES (leechcraft_poshuku_networkmonitor
	${QT_LIBRARIES}
	plugininterface
	xmlsettingsdialog
	)
INSTALL (TARGETS leechcraft_poshuku_networkmonitor DESTINATION ${LC_PLUGINS_DEST})
