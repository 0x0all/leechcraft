INCLUDE_DIRECTORIES (${AZOTH_INCLUDE_DIR}
	${CMAKE_CURRENT_BINARY_DIR}
	)
SET (AUTOPASTE_SRCS
	autopaste.cpp
	xmlsettingsmanager.cpp
	)
SET (AUTOPASTE_HEADERS
	autopaste.h
	xmlsettingsmanager.h
	)
QT4_WRAP_CPP (AUTOPASTE_MOC_SRCS ${AUTOPASTE_HEADERS})

	${AUTOPASTE_SRCS}
	${AUTOPASTE_MOC_SRCS}
	)

	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	)
