	${CMAKE_CURRENT_BINARY_DIR}
	${Boost_INCLUDE_DIR}
	${LEECHCRAFT_INCLUDE_DIR}
	)

	krigstask.cpp
	windowsmodel.cpp
	taskbarproxy.cpp
	)
#CreateTrs("krigstask" "en;ru_RU" COMPILED_TRANSLATIONS)
CreateTrsUpTarget("krigstask" "en;ru_RU" "${SRCS}" "${FORMS}" "")

	${COMPILED_TRANSLATIONS}
	${SRCS}
	)

	${QT_LIBRARIES}
	${LEECHCRAFT_LIBRARIES}
	${X11_X11_LIB}
	)
